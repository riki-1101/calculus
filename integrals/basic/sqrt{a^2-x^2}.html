\[ \int\sqrt{a^2-x^2}dx~~~(a\gt0) \]
\[ \begin{align} &\int\sqrt{a^2-x^2}dx\\ &=\int(x)'\sqrt{a^2-x^2}dx\\ &=x\sqrt{a^2-x^2}-\int x\cdot\frac{-2x}{2\sqrt{a^2-x^2}}dx\\ &=x\sqrt{a^2-x^2}-\int\frac{-x^2}{\sqrt{a^2-x^2}}dx\\ &=x\sqrt{a^2-x^2}-\int\frac{(a^2-x^2)-a^2}{\sqrt{a^2-x^2}}dx\\ &=x\sqrt{a^2-x^2}-\int\sqrt{a^2-x^2}dx+\int\frac{a^2}{\sqrt{a^2-x^2}}dx\\ \end{align} \]
\[ \begin{align} &\int\sqrt{a^2-x^2}dx\\ &=\frac{1}{2}\left(x\sqrt{a^2-x^2}+\int\frac{a^2}{\sqrt{a^2-x^2}}dx\right)\\ &=\frac{1}{2}\left(x\sqrt{a^2-x^2}+a^2\arcsin\frac{x}{a}\right)+C \end{align} \]
