\[ \int\frac{1}{a\sin^2x+b\cos^2x}dx~~~(a,b\gt0) \]
\[ \begin{align} &\int\frac{1}{a\sin^2x+b\cos^2x}dx\\ &=\int\frac{\sec^2x}{a\tan^2x+b}dx \end{align} \]
substitution \[ t=\tan x~~~~~dt=\sec^2xdx \]
\[ \begin{align} &\int\frac{\sec^2x}{a\tan^2x+b}dx\\ &=\int\frac{1}{at^2+b}dt\\ &=\frac{1}{a}\int\frac{1}{t^2+\frac{b}{a}}dt\\ &=\frac{1}{a}\cdot\sqrt{\frac{a}{b}}\arctan\sqrt{\frac{a}{b}}t+C\\ &=\frac{1}{\sqrt{ab}}\arctan\left(\sqrt{\frac{a}{b}}\tan x\right)+C\\ \end{align} \]
\[ \begin{align} &\int\frac{1}{a\sin^2x+b\cos^2x}dx\\ &=\frac{1}{\sqrt{ab}}\arctan\left(\sqrt{\frac{a}{b}}\tan x\right)+C \end{align} \]
