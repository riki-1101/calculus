\[ \int\frac{1}{x^2+x+1}dx \]
\[ \begin{align} &\int\frac{1}{x^2+x+1}dx\\ &=\int\frac{1}{\left(x+\frac{1}{2}\right)^2+\frac{3}{4}}dx\\ &=\int\frac{1}{\left(x+\frac{1}{2}\right)^2+\left(\frac{\sqrt{3}}{2}\right)^2}dx\\ &=\frac{2}{\sqrt{3}}\arctan\frac{2\left(x+\frac{1}{2}\right)}{\sqrt{3}}+C\\ &=\frac{2}{\sqrt{3}}\arctan\frac{2x+1}{\sqrt{3}}+C \end{align} \]
\[ \int\frac{1}{x^2+x+1}dx=\frac{2}{\sqrt{3}}\arctan\frac{2x+1}{\sqrt{3}}+C \]
