\[ \int\frac{1}{\sin x}dx \]
\[ \begin{align} &\int\frac{1}{\sin x}dx\\ &=\int\frac{1}{2\sin\frac{x}{2}\cos\frac{x}{2}}dx\\ &=\int\frac{1}{2\tan\frac{x}{2}\cos^2\frac{x}{2}}dx\\ &=\int\frac{1}{\tan\frac{x}{2}}\left(\tan\frac{x}{2}\right)'dx\\ &=\log\left|\tan\frac{x}{2}\right|+C \end{align} \]
\[ \int\frac{1}{\sin x}dx=\log\left|\tan\frac{x}{2}\right|+C \]
