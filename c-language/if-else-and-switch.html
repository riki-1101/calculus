if-else文

    処理;
}
else if(条件式){
    処理;

switch文

switch文は次のように記述する。

switch(式){
    case 定数式1:  文1;
                         break;
    case 定数式2:  文2;
                         break;
            :               :
    case 定数式n:  文n;
                         break;
    default:          文;
                         break;
}
